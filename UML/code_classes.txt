  +id: int
  +username: string
  +password: string
  +role: string
  +seConnecter()
  +seDeconnecter()
}

  +id: int
  +nom_produit: string
  +quantite: int
  +date_ajout: Date
  +utilisateur_id: int
  +ajouterProduit()
  +modifierQuantite()
  +supprimerProduit()
}

class Contact {
  +id: int
  +nom: string
  +email: string
  +num_tel: string
  +message: string
  +user_id: int
  +envoyerMessage()
}

class Produit {
  +id: int
  +nom: string

}

class Commande {
  +id: int
  +date_commande: Date
  +statut: string
  +total: float
  +user_id: int
  +validerCommande()
}
